Example

const { TelegramClient } = require('gramjs');
const { functions, types } = require('gramjs/tl');

(async () => {
    const client = new TelegramClient(name, apiId, apiHash);
    await client.start();

    const result = await client.invoke(new functions.account.VerifyPhoneRequest(
        phone_number='some string here',
        phone_code_hash='some string here',
        phone_code='some string here'
    ));
    console.log(result);
})();
